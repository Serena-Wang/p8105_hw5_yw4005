``` r
sim_normal <- function(n, mu, sigma) {
  
  sim_data = tibble(
    x = rnorm(n, mean = mu, sd = sigma),
  )
  
  sim_data %>% 
    summarize(
      mu_hat = mean(x),
      p_value = t.test(x, mu = 0, conf.level = 1-0.05) %>%
        tidy() %>%
        pull(p.value)
    )
}
```

``` r
sim_results_df = 
  expand_grid(
    sample_size = n,
    sigma = sigma,
    mu =  mu,
    iter = 1:5000
  ) %>% 
  mutate(
    estimate_df = pmap(list(sample_size, mu, sigma), sim_normal)
  ) %>% 
  unnest(estimate_df)
```
